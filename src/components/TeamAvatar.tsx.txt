import { cn } from "@/lib/utils";
import { accentRing } from "@/constants/team";
import type { TeamMember } from "@/types/onboarding";

const sizeMap = {
  sm: "h-11 w-11",
  md: "h-16 w-16",
  lg: "h-24 w-24",
} as const;

interface TeamAvatarProps {
  member: TeamMember;
  size?: keyof typeof sizeMap;
  active?: boolean;
  className?: string;
}

export function TeamAvatar({
  member,
  size = "md",
  active = false,
  className,
}: TeamAvatarProps) {
  return (
    <div className={cn("relative shrink-0", className)}>
      <img
        src={member.avatar}
        alt={`${member.name}, ${member.role}`}
        className={cn(
          "rounded-2xl object-cover ring-2 transition-all duration-300",
          sizeMap[size],
          active
            ? accentRing[member.accent]
            : "ring-border/60 opacity-90"
        )}
      />
      {active && (
        <span className="absolute -bottom-1 -right-1 flex h-4 w-4">
          <span className="absolute inline-flex h-full w-full animate-glow-pulse rounded-full bg-primary/70" />
          <span className="relative inline-flex h-4 w-4 rounded-full bg-primary ring-2 ring-background" />
        </span>
      )}
    </div>
  );
}
