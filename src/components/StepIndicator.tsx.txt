import { Check } from "lucide-react";
import { cn } from "@/lib/utils";
import { STEPS } from "@/constants/steps";

interface StepIndicatorProps {
  current: number;
  onSelect: (step: number) => void;
}

export function StepIndicator({ current, onSelect }: StepIndicatorProps) {
  const progress = (current / (STEPS.length - 1)) * 100;

  return (
    <nav aria-label="Onboarding progress" className="w-full">
      <div className="relative flex items-center justify-between">
        <div className="absolute left-0 right-0 top-5 h-0.5 -translate-y-1/2 rounded-full bg-border/70" />
        <div
          className="absolute left-0 top-5 h-0.5 -translate-y-1/2 rounded-full bg-gradient-to-r from-primary to-violet transition-all duration-500"
          style={{ width: `${progress}%` }}
        />
        {STEPS.map((step) => {
          const done = step.id < current;
          const activeStep = step.id === current;
          const reachable = step.id <= current;
          const Icon = step.icon;
          return (
            <button
              key={step.id}
              type="button"
              disabled={!reachable}
              onClick={() => reachable && onSelect(step.id)}
              className={cn(
                "relative z-10 flex flex-col items-center gap-2 focus:outline-none",
                reachable ? "cursor-pointer" : "cursor-not-allowed"
              )}
            >
              <span
                className={cn(
                  "flex h-10 w-10 items-center justify-center rounded-xl border transition-all duration-300",
                  activeStep &&
                    "border-primary bg-primary text-primary-foreground shadow-glow",
                  done &&
                    "border-primary/50 bg-primary/15 text-primary",
                  !activeStep &&
                    !done &&
                    "border-border bg-card text-muted-foreground"
                )}
              >
                {done ? (
                  <Check className="h-4 w-4" />
                ) : (
                  <Icon className="h-4 w-4" />
                )}
              </span>
              <span
                className={cn(
                  "hidden text-xs font-medium sm:block",
                  activeStep ? "text-foreground" : "text-muted-foreground"
                )}
              >
                {step.label}
              </span>
            </button>
          );
        })}
      </div>
    </nav>
  );
}
